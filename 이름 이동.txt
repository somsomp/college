#include <avr/io.h>
#include <util/delay.h>
#include <stdio.h>

int main(void){

	DDRA = 0xff;
	DDRC = 0xff;
	DDRF = 0x00;

	int N, R, A, C;
	char Key = 0x00;

	unsigned char  DATA[]= {		0b00101001,
									0b00101111,
									0b01101001,
									0b00101111,
									0b00000000,
									0b00111110,
									0b00100000,
									0b00000000,

									0b00001000,
									0b00011100,
									0b00110110,
									0b01100011,
									0b01001001,
									0b00001000,
									0b01111111,
									0b00000000,

									0b01000110,
									0b01101001,
									0b01001001,
									0b01100110,
									0b01000000,
									0b00000010,
									0b01111110,
									0b00000000	};

	while(1){
		while (Key == PINF){
			for(R = 0; R < 10; R++){
		      	for (N = 0; N < 8; N++){
		        	PORTA = ~(0x01 << N);
		        	PORTC = DATA[N];
		        	_delay_ms(1);	
		      	}
		  	}
		}
		
		Key = PINF;

		if (Key == 0x01) // аб
		  	for(A = 0; A < 8; A++) 
		   	    DATA[A] = (((DATA[A] & 0x01)<<7) | (DATA[A] >> 1));
		
		if (Key == 0x02) // ©Л
		    for(A = 0; A < 8; A++) 
		        DATA[A] = (DATA[A] << 1) | ((DATA[A] & 0x80) >>7);
		
		if (Key == 0x04){ // ╩С
		  	C = DATA[0];
		   	for(A = 0; A < 23; A++)
				DATA[A] = DATA[A+1];
			DATA[23] = C;
		}
		
		if (Key == 0x08){ // го
		    C = DATA[23];
		    for(A = 23; A > 0; A--)
				DATA[A] = DATA[A-1];
			DATA[0] = C;
		}
	}
}
